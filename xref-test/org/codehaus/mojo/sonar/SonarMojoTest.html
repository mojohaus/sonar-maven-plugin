90      @Test
91      public void shouldExportBinaries()
92          throws Exception
93      {
94          File localRepo = temp.newFolder();
95          final ArtifactRepository localRepository =
96              new DefaultArtifactRepository( "local", localRepo.toURI().toURL().toString(), new DefaultRepositoryLayout() );
97          File commonsIo = new File( localRepo, "commons-io/commons-io/2.4/commons-io-2.4.jar" );
98          FileUtils.forceMkdir( commonsIo.getParentFile() );
99          commonsIo.createNewFile();
100 
101         File baseDir = new File( "src/test/resources/org/codehaus/mojo/sonar/SonarMojoTest/sample-project" );
102         SonarMojo mojo = getMojo( baseDir );
103         mojo.setLocalRepository( localRepository );
104         mojo.execute();
105 
106         assertPropsContains( entry( "sonar.binaries", new File( baseDir, "target/classes" ).getAbsolutePath() ) );
107     }

131     @Test
132     public void shouldExportDependencies()
133         throws Exception
134     {
135         File localRepo = new File( "src/test/resources/org/codehaus/mojo/sonar/SonarMojoTest/repository" );
136         final ArtifactRepository localRepository =
137             new DefaultArtifactRepository( "local", localRepo.toURI().toURL().toString(), new DefaultRepositoryLayout() );
138         File baseDir = executeProject( "export-dependencies", localRepo );
139 
140         Properties outProps = readProps( "target/dump.properties" );
141         String libJson = outProps.getProperty( "sonar.maven.projectDependencies" );
142 
143         JSONAssert.assertEquals( "[{\"k\":\"commons-io:commons-io\",\"v\":\"2.4\",\"s\":\"compile\",\"d\":["
144             + "{\"k\":\"commons-lang:commons-lang\",\"v\":\"2.6\",\"s\":\"compile\",\"d\":[]}" + "]},"
145             + "{\"k\":\"junit:junit\",\"v\":\"3.8.1\",\"s\":\"test\",\"d\":[]}]", libJson, true );
146 
147         assertThat( outProps.getProperty( "sonar.java.binaries" ) ).isEqualTo( new File( baseDir, "target/classes" ).getAbsolutePath() );
148         assertThat( outProps.getProperty( "sonar.java.test.binaries" ) ).isEqualTo( new File( baseDir,
149                                                                                               "target/test-classes" ).getAbsolutePath() );
150     }

192     private File executeProject( String projectName, File localRepo )
193         throws Exception
194     {
195         ArtifactRepository artifactRepo =
196             new DefaultArtifactRepository( "local", localRepo.toURI().toURL().toString(), new DefaultRepositoryLayout() );
197 
198         File baseDir = new File( "src/test/resources/org/codehaus/mojo/sonar/SonarMojoTest/" + projectName );
199         SonarMojo mojo = getMojo( baseDir );
200         mojo.setLocalRepository( artifactRepo );
201         mojo.setLog( mockedLogger );
202 
203         mojo.execute();
204 
205         return baseDir;
206     }
207 
208     private void assertPropsContains( MapAssert.Entry... entries )
209         throws FileNotFoundException, IOException
210     {
211         assertThat( readProps( "target/dump.properties" ) ).includes( entries );
212     }
213 
214     private void assertGlobalPropsContains( MapAssert.Entry... entries )
215         throws FileNotFoundException, IOException
216     {
217         assertThat( readProps( "target/dump.properties.global" ) ).includes( entries );
218     }
219 
220     private Properties readProps( String filePath )
221         throws FileNotFoundException, IOException
222     {
223         FileInputStream fis = null;
224         try
225         {
226             File dump = new File( filePath );
227             Properties props = new Properties();
228             fis = new FileInputStream( dump );
229             props.load( fis );
230             return props;
231         }
232         finally
233         {
234             IOUtils.closeQuietly( fis );
235         }
236     }
