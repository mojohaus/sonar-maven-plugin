88      @Test
89      public void shouldExportBinaries()
90          throws Exception
91      {
92          File localRepo = temp.newFolder();
93          final ArtifactRepository localRepository =
94              new DefaultArtifactRepository( "local", localRepo.toURI().toURL().toString(), new DefaultRepositoryLayout() );
95          File commonsIo = new File( localRepo, "commons-io/commons-io/2.4/commons-io-2.4.jar" );
96          FileUtils.forceMkdir( commonsIo.getParentFile() );
97          commonsIo.createNewFile();
98  
99          File baseDir = new File( "src/test/resources/org/codehaus/mojo/sonar/SonarMojoTest/sample-project" );
100         SonarMojo mojo = getMojo( baseDir );
101         mojo.setLocalRepository( localRepository );
102         mojo.execute();
103 
104         assertPropsContains( entry( "sonar.binaries", new File( baseDir, "target/classes" ).getAbsolutePath() ) );
105     }

129     @Test
130     public void shouldExportDependencies()
131         throws Exception
132     {
133         File localRepo = new File( "src/test/resources/org/codehaus/mojo/sonar/SonarMojoTest/repository" );
134         final ArtifactRepository localRepository =
135             new DefaultArtifactRepository( "local", localRepo.toURI().toURL().toString(), new DefaultRepositoryLayout() );
136         File baseDir = executeProject( "export-dependencies", localRepo );
137 
138         Properties outProps = readProps( "target/dump.properties" );
139         String libJson = outProps.getProperty( "sonar.maven.projectDependencies" );
140 
141         JSONAssert.assertEquals( "[{\"k\":\"commons-io:commons-io\",\"v\":\"2.4\",\"s\":\"compile\",\"d\":["
142             + "{\"k\":\"commons-lang:commons-lang\",\"v\":\"2.6\",\"s\":\"compile\",\"d\":[]}" + "]},"
143             + "{\"k\":\"junit:junit\",\"v\":\"3.8.1\",\"s\":\"test\",\"d\":[]}]", libJson, true );
144 
145         assertThat( outProps.getProperty( "sonar.java.binaries" ) ).isEqualTo( new File( baseDir, "target/classes" ).getAbsolutePath() );
146         assertThat( outProps.getProperty( "sonar.java.test.binaries" ) ).isEqualTo( new File( baseDir,
147                                                                                               "target/test-classes" ).getAbsolutePath() );
148     }

190     private File executeProject( String projectName, File localRepo )
191         throws Exception
192     {
193         ArtifactRepository artifactRepo =
194             new DefaultArtifactRepository( "local", localRepo.toURI().toURL().toString(), new DefaultRepositoryLayout() );
195 
196         File baseDir = new File( "src/test/resources/org/codehaus/mojo/sonar/SonarMojoTest/" + projectName );
197         SonarMojo mojo = getMojo( baseDir );
198         mojo.setLocalRepository( artifactRepo );
199         mojo.setLog( mockedLogger );
200 
201         mojo.execute();
202 
203         return baseDir;
204     }
205 
206     private void assertPropsContains( MapAssert.Entry... entries )
207         throws FileNotFoundException, IOException
208     {
209         assertThat( readProps( "target/dump.properties" ) ).includes( entries );
210     }
211 
212     private Properties readProps( String filePath )
213         throws FileNotFoundException, IOException
214     {
215         FileInputStream fis = null;
216         try
217         {
218             File dump = new File( filePath );
219             Properties props = new Properties();
220             fis = new FileInputStream( dump );
221             props.load( fis );
222             return props;
223         }
224         finally
225         {
226             IOUtils.closeQuietly( fis );
227         }
228     }
