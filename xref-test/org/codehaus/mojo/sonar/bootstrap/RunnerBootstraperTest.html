53      @Before
54      public void setUp()
55          throws MojoExecutionException
56      {
57          MockitoAnnotations.initMocks( this );
58  
59          projectProperties = new Properties();
60          when(
61                mavenProjectConverter.configure( anyListOf( MavenProject.class ), any( MavenProject.class ),
62                                                 any( Properties.class ) ) ).thenReturn( projectProperties );
63          List<Object> extensions = new LinkedList<Object>();
64          extensions.add( new Object() );
65          when( extensionsFactory.createExtensions() ).thenReturn( extensions );
66          when( extensionsFactory.createExtensionsWithDependencyProperty() ).thenReturn( extensions );
67  
68          when( runner.mask( anyString() ) ).thenReturn( runner );
69          when( runner.unmask( anyString() ) ).thenReturn( runner );
70          runnerBootstrapper =
71              new RunnerBootstrapper( log, session, securityDispatcher, runner, mavenProjectConverter, extensionsFactory );
72      }
73  
74      @Test
75      public void testSQ52()
76          throws MojoExecutionException, IOException
77      {
78          when( runner.serverVersion() ).thenReturn( "5.2" );
79          runnerBootstrapper.execute();
80  
81          verifyCommonCalls();
82  
83          // no extensions, mask or unmask
84          verifyNoMoreInteractions( runner );
