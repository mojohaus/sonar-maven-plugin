37  /**
38   * {@link MavenPluginManager} helper to deal with API changes between Maven 3.0.x and 3.1.x. Inspired from
39   * maven-reporting-exec
40   * 
41   * @since 2.1
42   */
43  public interface MavenPluginManagerHelper
44  {
45      PluginDescriptor getPluginDescriptor( Plugin plugin, MavenSession session )
46          throws PluginResolutionException, PluginDescriptorParsingException, InvalidPluginDescriptorException;
47  
48      void setupPluginRealm( PluginDescriptor pluginDescriptor, MavenSession session, ClassLoader parent,
49                             List<String> imports )
50          throws PluginResolutionException, PluginContainerException;
51  
52  }
