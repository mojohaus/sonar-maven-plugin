20  package org.codehaus.mojo.sonar.bootstrap;
21  
22  import org.apache.maven.plugin.MojoExecutionException;
23  import org.apache.maven.plugin.logging.Log;
24  
25  class ExceptionHandling
26  {
27  
28      private ExceptionHandling()
29      {
30          // Hide public constructor
31      }
32  
33      static RuntimeException handle( Exception e, Log log )
34          throws MojoExecutionException
35      {
36          Throwable source = e;
37          if ( "org.sonar.runner.impl.RunnerException".equals( e.getClass().getName() ) && e.getCause() != null )
38          {
39              source = e.getCause();
40          }
41          throw new MojoExecutionException( source.getMessage(), source );
42      }
43  
44      static RuntimeException handle( String message, Log log )
45          throws MojoExecutionException
46      {
47          return handle( new MojoExecutionException( message ), log );
48      }
49  }
