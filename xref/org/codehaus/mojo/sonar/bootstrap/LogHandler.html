1   package org.codehaus.mojo.sonar.bootstrap;
2   
3   import org.apache.maven.plugin.logging.Log;
4   
5   import org.sonar.runner.api.LogOutput;
6   
7   public class LogHandler
8       implements LogOutput
9   {
10      private Log mavenLog;
11  
12      public LogHandler( Log mavenLog )
13      {
14          this.mavenLog = mavenLog;
15      }
16  
17      @Override
18      public void log( String log, Level level )
19      {
20          switch ( level )
21          {
22              case TRACE:
23                  mavenLog.debug( log );
24                  break;
25              case DEBUG:
26                  mavenLog.debug( log );
27                  break;
28              case INFO:
29                  mavenLog.info( log );
30                  break;
31              case WARN:
32                  mavenLog.warn( log );
33                  break;
34              case ERROR:
35                  mavenLog.error( log );
36                  break;
37          }
38      }
39  }
