20  package org.codehaus.mojo.sonar.bootstrap;
21  
22  import org.apache.maven.model.Plugin;
23  import org.apache.maven.project.MavenProject;
24  import org.codehaus.plexus.util.xml.Xpp3Dom;
25  
26  /**
27   * An utility class to manipulate Maven concepts
28   *
29   * @since 1.10
30   */
31  public final class MavenUtils
32  {
33  
34      private static final String MAVEN_COMPILER_PLUGIN = "maven-compiler-plugin";
35  
36      public static final String GROUP_ID_APACHE_MAVEN = "org.apache.maven.plugins";
37  
38      public static final String GROUP_ID_CODEHAUS_MOJO = "org.codehaus.mojo";
39  
40      private MavenUtils()
41      {
42          // utility class with only static methods
43      }
44  
45      /**
46       * Returns the version of Java used by the maven compiler plugin
47       *
48       * @param pom the project pom
49       * @return the java version
50       */
51      public static String getJavaVersion( MavenProject pom )
52      {
53          MavenPlugin compilerPlugin = MavenPlugin.getPlugin( pom, GROUP_ID_APACHE_MAVEN, MAVEN_COMPILER_PLUGIN );
54          if ( compilerPlugin != null )
55          {
56              return compilerPlugin.getParameter( "target" );
57          }
58          return null;
59      }
60  
61      public static String getJavaSourceVersion( MavenProject pom )
62      {
63          MavenPlugin compilerPlugin = MavenPlugin.getPlugin( pom, GROUP_ID_APACHE_MAVEN, MAVEN_COMPILER_PLUGIN );
64          if ( compilerPlugin != null )
65          {
66              return compilerPlugin.getParameter( "source" );
67          }
68          return null;
69      }
70  
71      /**
72       * @return source encoding
73       */
74      public static String getSourceEncoding( MavenProject pom )
75      {
76          return pom.getProperties().getProperty( "project.build.sourceEncoding" );
77      }
78  
79      /**
80       * Search for a configuration setting of an other plugin for a configuration setting.
81       *
82       * @todo there should be a better way to do this
83       * @param project the current maven project to get the configuration from.
84       * @param pluginId the group id and artifact id of the plugin to search for
85       * @param optionName the option to get from the configuration
86       * @param defaultValue the default value if the configuration was not found
87       * @return the value of the option configured in the plugin configuration
88       */
89      public static String getPluginSetting( MavenProject project, String pluginId, String optionName,
90                                             String defaultValue )
91      {
92          Xpp3Dom dom = getPluginConfigurationDom( project, pluginId );
93          if ( dom != null && dom.getChild( optionName ) != null )
94          {
95              return dom.getChild( optionName ).getValue();
96          }
97          return defaultValue;
98      }
99  
100     /**
101      * Search for the configuration Xpp3 dom of an other plugin.
102      *
103      * @todo there should be a better way to do this
104      * @param project the current maven project to get the configuration from.
105      * @param pluginId the group id and artifact id of the plugin to search for
106      * @return the value of the option configured in the plugin configuration
107      */
108     private static Xpp3Dom getPluginConfigurationDom( MavenProject project, String pluginId )
109     {
110 
111         Plugin plugin = project.getBuild().getPluginsAsMap().get( pluginId );
112         if ( plugin != null )
113         {
114             // TODO: This may cause ClassCastExceptions eventually, if the dom impls differ.
115             return (Xpp3Dom) plugin.getConfiguration();
116         }
117         return null;
118     }
119 }
