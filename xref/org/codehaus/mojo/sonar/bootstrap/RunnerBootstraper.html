46  /**
47   * Configure properties and bootstrap using SonarQube runner API (need SQ 4.3+)
48   */
49  public class RunnerBootstraper
50  {
51  
52      private final RuntimeInformation runtimeInformation;
53  
54      private final Log log;
55  
56      private final MavenSession session;
57  
58      private final LifecycleExecutor lifecycleExecutor;
59  
60      private final ArtifactFactory artifactFactory;
61  
62      private final ArtifactRepository localRepository;
63  
64      private final ArtifactMetadataSource artifactMetadataSource;
65  
66      private final ArtifactCollector artifactCollector;
67  
68      private final DependencyTreeBuilder dependencyTreeBuilder;
69  
70      private final MavenProjectBuilder projectBuilder;
71  
72      private final SecDispatcher securityDispatcher;
73  
74      private ServerMetadata server;
75  
76      public RunnerBootstraper( RuntimeInformation runtimeInformation, Log log,
77                                MavenSession session, LifecycleExecutor lifecycleExecutor,
78                                ArtifactFactory artifactFactory, ArtifactRepository localRepository,
79                                ArtifactMetadataSource artifactMetadataSource, ArtifactCollector artifactCollector,
80                                DependencyTreeBuilder dependencyTreeBuilder, MavenProjectBuilder projectBuilder,
81                                SecDispatcher securityDispatcher, ServerMetadata server )
82      {
83          this.runtimeInformation = runtimeInformation;
84          this.log = log;
85          this.session = session;
86          this.lifecycleExecutor = lifecycleExecutor;
87          this.artifactFactory = artifactFactory;
88          this.localRepository = localRepository;
89          this.artifactMetadataSource = artifactMetadataSource;
90          this.artifactCollector = artifactCollector;
91          this.dependencyTreeBuilder = dependencyTreeBuilder;
92          this.projectBuilder = projectBuilder;
93          this.securityDispatcher = securityDispatcher;
94          this.server = server;
95      }
96  
97      public void execute()
98          throws IOException, MojoExecutionException
99      {
100         try
101         {
102             EmbeddedRunner runner =
103                 EmbeddedRunner.create()
104                               .setApp( "Maven", runtimeInformation.getMavenVersion() )
105                               .addProperties( session.getSystemProperties() );
106             // Exclude log implementation to not conflict with Maven 3.1 logging impl
107             runner.mask( "org.slf4j.LoggerFactory" )
108                   // Include slf4j Logger that is exposed by some Sonar components
109                   .unmask( "org.slf4j.Logger" ).unmask( "org.slf4j.ILoggerFactory" )
110                   // Exclude other slf4j classes
111                   // .unmask("org.slf4j.impl.")
112                   .mask( "org.slf4j." )
113                   // Exclude logback
114                   .mask( "ch.qos.logback." )
115                   .mask( "org.sonar." )
116                   // Guava is not the same version in SonarQube classloader
117                   .mask( "com.google.common" )
118                   // Include everything else
119                   .unmask( "" );
120             runner.addExtensions( session, log, lifecycleExecutor, projectBuilder );
121             if ( !server.supportsNewDependencyProperty() )
122             {
123                 runner.addExtensions( artifactFactory, localRepository, artifactMetadataSource, artifactCollector,
124                                       dependencyTreeBuilder );
125             }
126             if ( log.isDebugEnabled() )
127             {
128                 runner.setProperty( "sonar.verbose", "true" );
129             }
130             runner.addProperties( collectProperties() );
131 
132             // Secret property to manage backward compatibility on SQ side (see ProjectScanContainer)
133             runner.setProperty( "sonar.mojoUseRunner", "true" );
134 
135             runner.execute();
136         }
137         catch ( Exception e )
138         {
139             throw ExceptionHandling.handle( e, log );
140         }
141     }
142 
143     private Properties collectProperties()
144         throws MojoExecutionException
145     {
146         Properties props =
147             new MavenProjectConverter( log, server.supportsFilesAsSources(),
148                                        new DependencyCollector( dependencyTreeBuilder, artifactFactory,
149                                                                 localRepository, artifactMetadataSource,
150                                                                 artifactCollector ) ).configure( session.getProjects(),
151                                                                                                  session.getTopLevelProject(),
152                                                                                                  session.getUserProperties() );
153         props.putAll( decryptProperties( props ) );
154 
155         return props;
156     }
157 
158     public Properties decryptProperties( Properties properties )
159     {
160         Properties newProperties = new Properties();
161         try
162         {
163             for ( String key : properties.stringPropertyNames() )
164             {
165                 if ( key.contains( ".password" ) )
166                 {
167                     decrypt( properties, newProperties, key );
168                 }
169             }
170         }
171         catch ( Exception e )
172         {
173             log.warn( "Unable to decrypt properties", e );
174         }
175         return newProperties;
176     }
177 
178     private void decrypt( Properties properties, Properties newProperties, String key )
179     {
180         try
181         {
182             String decrypted = securityDispatcher.decrypt( properties.getProperty( key ) );
183             newProperties.setProperty( key, decrypted );
184         }
185         catch ( SecDispatcherException e )
186         {
187             log.debug( "Unable to decrypt property " + key, e );
188         }
189     }
190 }
