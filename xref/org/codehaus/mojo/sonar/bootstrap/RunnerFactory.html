3   import org.apache.maven.rtinfo.RuntimeInformation;
4   
5   import org.sonar.runner.api.LogOutput;
6   import org.sonar.runner.api.EmbeddedRunner;
7   import org.apache.maven.execution.MavenSession;
8   
9   public class RunnerFactory
10  {
11  
12      private final LogOutput logOutput;
13  
14      private final RuntimeInformation runtimeInformation;
15  
16      private final MavenSession session;
17  
18      private final boolean debugEnabled;
19  
20      public RunnerFactory( LogOutput logOutput, boolean debugEnabled, RuntimeInformation runtimeInformation,
21                            MavenSession session )
22      {
23          this.logOutput = logOutput;
24          this.runtimeInformation = runtimeInformation;
25          this.session = session;
26          this.debugEnabled = debugEnabled;
27      }
28  
29      public EmbeddedRunner create()
30      {
31          EmbeddedRunner runner = EmbeddedRunner.create( logOutput );
32          runner.setApp( "Maven", runtimeInformation.getMavenVersion() );
33          runner.addGlobalProperties( session.getSystemProperties() );
34          // Secret property to manage backward compatibility on SQ side (see ProjectScanContainer)
35          runner.setGlobalProperty( "sonar.mojoUseRunner", "true" );
36          if ( debugEnabled )
37          {
38              runner.setGlobalProperty( "sonar.verbose", "true" );
39          }
40  
41          return runner;
42      }
43  }
