1   package org.codehaus.mojo.sonar.bootstrap;

3   import org.apache.maven.execution.MavenSession;
4   import org.apache.maven.rtinfo.RuntimeInformation;
5   import org.sonar.runner.api.EmbeddedRunner;
6   import org.sonar.runner.api.LogOutput;
7   
8   import java.util.Properties;
9   
10  public class RunnerFactory
11  {
12  
13      private final LogOutput logOutput;
14  
15      private final RuntimeInformation runtimeInformation;
16  
17      private final MavenSession session;
18  
19      private final boolean debugEnabled;
20  
21      public RunnerFactory( LogOutput logOutput, boolean debugEnabled, RuntimeInformation runtimeInformation,
22                            MavenSession session )
23      {
24          this.logOutput = logOutput;
25          this.runtimeInformation = runtimeInformation;
26          this.session = session;
27          this.debugEnabled = debugEnabled;
28      }
29  
30      public EmbeddedRunner create()
31      {
32          EmbeddedRunner runner = EmbeddedRunner.create( logOutput );
33          runner.setApp( "Maven", runtimeInformation.getMavenVersion() );
34  
35          runner.addGlobalProperties( createGlobalProperties() );
36  
37          // Secret property to manage backward compatibility on SQ side (see ProjectScanContainer)
38          runner.setGlobalProperty( "sonar.mojoUseRunner", "true" );
39          if ( debugEnabled )
40          {
41              runner.setGlobalProperty( "sonar.verbose", "true" );
42          }
43  
44          return runner;
45      }
46  
47      private Properties createGlobalProperties()
48      {
49          Properties p = new Properties();
50          p.putAll( session.getTopLevelProject().getProperties() );
51          p.putAll( session.getSystemProperties() );
52          p.putAll( session.getUserProperties() );
53          return p;
54      }
55  }
